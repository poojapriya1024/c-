class Solution {
public:
    string mergeAlternately(string word1, string word2) {

        int len1 = word1.size(); 
        int len2 = word2.size();

        string merged = "";

        int i=0,j=0;
        while(i<len1 && j<len2)
        {
            merged+= word1[i++];
            merged+= word2[j++];
        }

        while(i<len1)
            merged+= word1[i++];

        while(j<len2)
            merged+= word2[j++];

        return merged;
    }
};
